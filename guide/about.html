# 基本结构

代码

<body>
    <div class="ui-page-group">
        <div class="ui-page ui-page-current ">
            <!-- 页头，底部导航位置 -->
            <include file="pub/_tabbar" />
            <!-- 页面内容区开始-->
            <div class="ui-content">
                <!-- 页面主体内容 -->
            </div>

            <!-- 弹窗 -->
            <div class="panel-overlay">
                
            </div>
        </div>
    </div>
</body>
